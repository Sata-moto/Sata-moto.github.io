ODT 学习笔记

发布于 2020-02-28 | 分类于 知识点 、 数据结构 | 11分钟 | 2330字数

前言：

\quad\quad 最近被各种奇葩数据结构给虐得惨无人形...

\quad\quad 然后意外地看到了这道题：CF896C Willem, Chtholly and Seniorious

\quad\quad Get到了一种神（kai）奇（che）的数据结构...ODT(old driver tree)

\quad\quad 写一篇学习笔记加以纪念...

正文：

\quad\quad ODT:老司机树（Old Driver Tree），又称珂朵莉树（Chtholly Tree），起源于某位巨佬的CF896C的题解...

\quad\quad 这位巨佬在用线段树爆切了上面那题后，又提出了一种神奇的暴力骗分数据结构：ODT...

\quad\quad ODT适用于解决带有区间染色，区间赋值（也可看做区间染色）类型的问题，要求数据随机...（数据不随机时，它的复杂度是假的，所以只能作为暴力骗分算法看待，不能作为正解）

\quad\quad 我们首先拉出一个问题：在一条长为L的线段上任给两个点，问这两个点之间的距离的期望值...

\quad\quad 很容易算出来期望值为L/3，证明如下:

\quad\quad 分布函数为F(x)=2x/L−(x/L)2F(x)=2x/L-(x/L)^2F(x)=2x/L−(x/L)2

\quad\quad 分布密度函数为f(x)=2/L−2x/L2f(x)=2/L-2x/L^2f(x)=2/L−2x/L2

\quad\quad 故期望为E(x)=∫xf(x)dx=L/3E(x)=∫xf(x)dx=L/3E(x)=∫xf(x)dx=L/3

\quad\quad 也就是说，每次染色可以把1/3的区间上的颜色变为同一种...

\quad\quad 由此看来，在进行若干次随机的区间染色操作后，区间中的线段数目会比较少...

\quad\quad 基于 区间中的颜色种数会比较少 这一性质，我们可以直接把同颜色的块缩成一个点，然后对于这不多的点暴力处理问题...

\quad\quad 如何维护这些不多的点？我们一般使用平衡树...

\quad\quad 当然，使用数组或者链表也可以，但是数组难以高效地插入删除，链表难以高效地查询...

\quad\quad 使用哪种平衡树呢？秉承着能偷懒就偷懒的骗分原则，我们选择set...

\quad\quad 以下给出本题中ODT的几大操作（网上关于ODT的写法大体都差不多，所以以下代码与其它人代码相似，绝非巧合）

InitInitInit （定义和初始化）：

\quad\quad 定义一个ODT很简单，我们只需要一个结构体和一个set就可以搞定，因为我们set维护的是某一颜色块缩成的点，所以我们结构体应存储当前颜色块的左右端点和颜色，然后直接放到set里就好了...

#define int long long
//不想考虑全文的int溢出的话，就偷个懒吧

struct node
{
	int l, r;//左右端点
	mutable int key;//颜色
 	//mutable可以解除const状态，set中元素如果不带mutable，是不能修改的...

	node (int a, int b, int c)
	{
		l = a, r = b, key = c;
	}
	node () { }
	//构造函数

	bool operator < (const node & other) const
	{
		return l < other.l;
	}
	//因为是set，所以要重载运算符    
};
set <node > s;

\quad\quad 初始化也很简单，最开始的时候，每个节点作为单独的颜色块插入就好了，不用提前缩点，毕竟长度为1/3L的区间染色后缩点优化的复杂度，初始时缩点所优化的复杂度可以忽略不计...

for(register int k = 1; k <= n; k++)
	s.insert(node(k, k, rnd() % maxn + 1));
//rnd() % maxn + 1是本题中的初始点权

SplitSplitSplit（分裂）:

\quad\quad 当我们把某个颜色块合并成某个点后，如果我们以后的某个操作并没有完全覆盖该颜色块，我们就需要把该颜色块从某个点处分成两半...

\quad\quad SplitSplitSplit函数的作用，就是把某个颜色块节点分成两半，并返回后面一半的迭代器...

//把某个颜色块[l,r]分成[l,wz-1]和[wz,r]
IT split(int wz)
{
	IT it = s.lower_bound(node(wz, 0, 0));
	//二分查找第一个大于等于该点的颜色段左端点

	if(it != s.end() && it->l == wz) return it;
	//如果这个点本来就是某个颜色块的左端点
	//那么我们就不需要分裂这个点了

	--it;
	//否则，需要分裂的颜色段一定是--it

	int L = it->l, R = it->r, key = it->key;
	//提取需要分裂的颜色端的信息 

	s.erase(it);
	//删掉这个颜色端 
	
	s.insert(node(L, wz - 1, key));
	return s.insert(node(wz, R, key)).first;
	//加入 [l,wz-1]和[wz,r]两个新的颜色端并且返回后面这个颜色端的迭代器 
}

AssignAssignAssign（合并）[op=2op=2op=2]:

\quad\quad 这就是ODT能够水题的基础操作...

\quad\quad 把一段区间里的所有玩意全部染上某种颜色...

\quad\quad 这里介绍set内置的一个函数：

\quad\quad void erase (iterator first, iterator last)，可以删除[first,last)区间的所有玩意...

\quad\quad 代码很简单：

void assign(int l, int r, int key)
{
	IT R = split(r + 1), L = split(l);
	//先把左右端点从它们在的颜色段里割出来...
	//注意，此处要先割R,再割L,因为割R的时候有可能干扰先割出来的L 
	
	s.erase(L, R);
	//删掉区间内的所有东西（左闭右开） 
	
	s.insert(node(l, r, key));
	//加入一个新的颜色段 
}

KthKthKth（区间第k小）[op=3op=3op=3]:


//没啥好说的，反正点很少，割出来暴力排序即可 
int Kth(int l, int r, int x)
{
	IT R = split(r + 1), L = split(l);
	vector <pair <int , int > > v;

	for(; L != R; ++L) v.push_back(mp(L->key, L->r - L->l + 1));
	sort(v.begin(), v.end());

	for(int k = 0; k < (int )v.size(); k++)
	{
		x -= v[k].second;
		if(x <= 0) return v[k].first;
	}
	return -1ll;
}

SumSumSum（区间x次方和）[op=4op=4op=4]:

\quad\quad 引用某数据结构毒瘤在网课上的发言：数据结构就不是能维护这样的东西的东西...

\quad\quad 所以暴力吧...

//暴力即可，x次方要写个快速幂 
int Sum(int l, int r, int x, int y)
{
	IT R = split(r + 1), L = split(l);
	int ret = 0;

	for(; L != R; ++L)
		ret = (ret + (L->r - L->l + 1) * quick_pow(L->key, x, y) ) % y;
	return ret;
}

\quad\quad 综上，可以看出：ODT的精髓就在于合并，合并完了之后随意暴力乱搞，什么O(n)O(n)O(n)（扫一遍）,O(nlogn)O(nlogn)O(nlogn)（排个序）,乃至O(n2)O(n^2)O(n2)（乱搞），的单次复杂度都是可以接受的了...

\quad\quad 据说数据全随机的情况下，复杂度可以证明是O(mlogn)O(mlogn)O(mlogn)...

\quad\quad 实际跑起来的速度，在不卡ODT的题目里，是相当优秀的...

\quad\quad 下面附例题完整代码：

#include <bits/stdc++.h>
#define mp make_pair
#define int long long
#define IT set <node > :: iterator

using namespace std;

const int N = 1e5 + 10;
const int MOD = 1e9 + 7;

int quick_pow(int x, int y, int P)
{
	x %= P;
	int mult = x, ans = 1;

	while(y)
	{
		if(y & 1)
			ans *= mult, ans %= P;

		mult *= mult, mult %= P, y >>= 1;
	}

	return ans % P;
}

struct node
{
	int l, r;
	mutable int key;

	node (int a, int b, int c)
	{
		l = a, r = b, key = c;
	}
	node () { }

	bool operator < (const node & other) const
	{
		return l < other.l;
	}
};
set <node > s;

int n, m, maxn;
int seed;

int rnd()
{
	int ret = seed;
	seed = (seed * 7 + 13) % MOD;
	return ret;
}

IT split(int wz)
{
	IT it = s.lower_bound(node(wz, 0, 0));
	if(it != s.end() && it->l == wz) return it;

	--it;
	int L = it->l, R = it->r, key = it->key;
	s.erase(it),s.insert(node(L, wz - 1, key));
	return s.insert(node(wz, R, key)).first;
}
void assign(int l, int r, int key)
{
	IT R = split(r + 1), L = split(l);
	s.erase(L, R),s.insert(node(l, r, key));
}

void Add(int l, int r, int x)
{
	IT R = split(r + 1), L = split(l);
	for(; L != R; ++L) L->key += x;
}

int Kth(int l, int r, int x)
{
	IT R = split(r + 1), L = split(l);
	vector <pair <int , int > > v;

	for(; L != R; ++L) v.push_back(mp(L->key, L->r - L->l + 1));
	sort(v.begin(), v.end());

	for(int k = 0; k < (int )v.size(); k++)
	{
		x -= v[k].second;
		if(x <= 0) return v[k].first;
	}
	return -1ll;
}

int Sum(int l, int r, int x, int y)
{
	IT R = split(r + 1), L = split(l);
	int ret = 0;

	for(; L != R; ++L)
		ret = (ret + (L->r - L->l + 1) * quick_pow(L->key, x, y) ) % y;
	return ret;
}

signed main()
{
	scanf("%lld %lld %lld %lld", &n, &m, &seed, &maxn);
	for(register int k = 1; k <= n; k++)
		s.insert(node(k, k, rnd() % maxn + 1));
	for(register int k = 1; k <= m; k++)
	{
		int op = (rnd() % 4) + 1, l = (rnd() % n) + 1, r = (rnd() % n) + 1, x;
		if(l > r) swap(l, r);

		if(op == 3)
			x = (rnd() % (r - l + 1)) + 1;
		else
			x = (rnd() % maxn) + 1;

		if(op == 1)
			Add(l, r, x);
		if(op == 2)
			assign(l, r, x);
		if(op == 3)
			printf("%lld\n", Kth(l, r, x));
		if(op == 4)
		{
			int y = (rnd() % maxn) + 1;
			printf("%lld\n", Sum(l, r, x, y));
		}
	}

	return 0;
}
