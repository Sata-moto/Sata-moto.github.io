正文：

\quad\quad 学习SplaySplaySplay之前，需要一个前置知识点：二叉查找树

\quad\quad 什么是二叉查找树？顾名思义：二叉的查找树...

\quad\quad 具体而言：对于树中的任意节点，它左子树中的所有节点都比它小，右子树中的所有节点都比它大（递归定义）...

\quad\quad 为什么叫它查找树呢？因为我们可以很容易地查找出树中第K大的元素是多少，以及值为X的元素排名第几...

\quad\quad 实现方式是：我们维护一个size[N]size[N]size[N]数组，记录每个节点的子树大小，查找的时候，从根节点开始，比较当前sizesizesize与所查找的信息的大小，然后一步步向下走就好了。这里不多讲，因为后面SplaySplaySplay中的查找操作也是这么实现的...

\quad\quad 同一种二叉查找树有很多不同的形态...

\quad\quad 比如说由1 2 3 4 5构成的二叉查找树...可以是这样的：

\quad\quad 也可以是这样的：

\quad\quad 我们知道：均摊来讲，一棵树的高仅仅只有lognlognlogn，这使得依赖于树高的算法拥有极优的复杂制度...

\quad\quad 但作为一种形态不固定的树形结构，二叉查找树的单次操作复杂度，很容易被非随机数据卡到O(n)O(n)O(n)...

\quad\quad 于是，我们神奇的SplaySplaySplay，就应运而生...

\quad\quad SplaySplaySplay，也叫伸展树，本质上类似二叉查找树（后面会发现，在处理区间问题时，千万千万不能把它当做二叉查找树）...但与一般二叉查找树不同，它有一个很神奇的操作：伸展（SplaySplaySplay）...

\quad\quad 它主要用来处理（二叉查找树都可以处理的）排名问题，和一些（FHQ_treap可以解决的）区间问题...比如说：进行一番动态插入删除只后，树中排名为K的节点是多少，数值为x的节点，排名多少，某个节点的前驱，后继等....

\quad\quad 下面是解决排名问题的权值SplaySplaySplay

\quad\quad 先给出它的变量列表：

  const int N=100010;

  int num[N];//编号为i的点，代表的数值是多少...
  int cnt[N];//编号为i的点，代表的数值出现了多少次...
  int fa[N],son[N][2];//编号为i的点，父亲与左右儿子编号各是多少...
  int size[N];//以编号为i的点为根的子树的节点的cnt的和是多少...

  int tot;//计数工具：最后一个新建的点的编号是多少
  int root;//Splay的根的编号是多少

\quad\quad 需要注意的是，我们的所有数组类型的变量，都是关于节点编号的映射，不可以把节点数值带入...

\quad\quad 还有一点需要注意的是：每个节点的出现次数可能不为一...

\quad\quad （如果你已经掌握二叉查找树的内容，以上变量都不难理解）

\quad\quad 下面是详细的旋转步骤：

	1：设旋转的节点为x点，其父节点为y点，祖父节点为z点...
	2：设x是y的m儿子（m为0/1，代表左/右儿子，下同），y是z的n儿子...
	3: 把x的m^1 (0^1=1/1^1=0)儿子变为y的m儿子
	4：把y变为x的m^1儿子
  	5：把x变为z的n儿子
	6：更新操作后x,y的size值（为什么其他点不用更新？）

\quad\quad 手动模拟以上步骤，容易发现，旋转后SplaySplaySplay的结构特性并未破坏，但点与点之间的“亲戚关系 ” ，发生了改变...

\quad\quad 下面是代码描述，从上到下分别于上面的步骤相对应：

  void rotate(int x)
  {
      int y = fa[x], z = fa[y];
      
      int m = find(x) , n = find(f);

      connect(son[x][m ^ 1], y, m);
      
      connect(y, x, m ^ 1);
      
      connect(x, z, n);
      
      push_up(y), push_up(x);
  }

\quad\quad 注意要点：连接顺序和更新顺序...

\quad\quad find(),connect(),push_up()find(),connect(),push\_up()find(),connect(),push_up()是三个辅助函数，作用分别为：得到某个节点是它父亲的什么儿子（0/1分别对应左/右儿子）；连接某两个点；更新某个点的sizesizesize值...代码如下：

  }

  void connect(int x, int y, int w_son)

      if(y) son[y][w_son] = x;
      if(x) fa[x] = y;

  void push_up(int x)

\quad\quad 注意要点：connectconnectconnect函数一定不能把0点与其他点连起来（在之后的操作中是可能调用connect(X,0,X)connect(X,0,X)connect(X,0,X)的）...

\quad\quad rotaterotaterotate有什么用？它是SplaySplaySplay操作之精髓：[Splay][Splay][Splay]的必要工具...

SplaySplaySplay（伸展）:

\quad\quad 很难说清SplaySplaySplay操作的意义是什么...不过一般的理解是：它可以将某个点从某个地方一路旋转到根节点处，并在一路旋转的过程中使整棵树更加平衡...

\quad\quad 至于它到底是怎么做到平衡的...Emmm...

\quad\quad 让我们直接看一看SplaySplaySplay的代码（这里是直接旋到根了）：

  void Splay(int x)
  {
      for(int f; (f = fa[x]) != 0; rotate(x))
          if(fa[f]) rotate(find(f) == find(x) ? f : x);

      root = x;
  }

\quad\quad 大体思路是，把某个点一直向上旋...直到旋到根节点...

\quad\quad 不过，在这个过程中有一个特殊情况：当旋转节点，旋转节点父节点，旋转节点祖父节点“三点一线”（旋转节点是父节点的m儿子，父节点也是祖父节点的m儿子）时，要先转一下父节点，再转一下当前节点...

\quad\quad 代码十分简洁，不过理解起来不是很方便...

\quad\quad 可以看做把两次rotaterotaterotate看做一组，如果三点一线就先父再子，否则转两次子...

\quad\quad 最后不要忘了把根赋给SplaySplaySplay上来的节点...

\quad\quad 接下来就直接上代码了...都是些常用的函数...

FindFindFind（得到某个点的排名，并将其Splay上来）:

  //递推版本（递归常数大，但更好写）
  int Find(int x)
  {
      int now = root, ans = 0;

      while(true)
      {
          if(x < num[now])
          {
              now = son[now][0];
              continue;
          }

          ans += size[son[now][0]];

          if(num[now] == x)
          {
              Splay(now);
              return ans + 1;
          }

          ans += cnt[now];

          now = son[now][1];
      }
  }

RankRankRank（得到排行为x的节点的值）:

  int Rank(int rank)
  {
      int now = root;

      while(1)
      {
          if(rank <= size[son[now][0]])
          {
              now = son[now][0];
              continue;
          }
          rank -= size[son[now][0]];

          if(rank <= cnt[now])
          {
              Splay(now);
              return num[now];
          }
          rank -= cnt[now];

          now = son[now][1];
      }
  }

InsertInsertInsert（插入）:

  void Insert(int x)
  {
      if(root == 0)
          root = ++tot, size[tot] = cnt[tot] = 1, num[tot] = x;
      else
      {
          int now = root, f = 0;

          while(true)
          {
              if(num[now] == x)
              {
                  cnt[now]++;
                  push_up(now), push_up(f);
                  Splay(now);

                  return ;
              }

              f = now, now = son[now][x > num[now]];

              if(!now)
              {
                  num[++tot] = x, size[tot] = cnt[tot] = 1;
                  fa[tot] = f, son[f][x > num[f]] = tot;
                  push_up(f), Splay(tot);

                  return ;
              }
          }
      }
  }

DeleteDeleteDelete（删除）:

  void Delete(int x)
  {
      Find(x);

      int now = root;

      if(cnt[now] > 1) cnt[now]--, size[now]--;
      else if(!son[now][0] && !son[now][1])
          root = 0, clear(now);
      else if(!son[now][0] && son[now][1])
          fa[root = son[now][1]] = 0, clear(now);
      else if(son[now][0] && !son[now][1])
          fa[root = son[now][0]] = 0, clear(now);
      else
      {
          int pre = Pre(), rt = root, rson = son[root][1];

          Splay(pre), connect(rson, pre, 1);
          clear(rt), push_up(root);
      }
  }

PrePrePre（得到根的前驱的编号）:

  int Pre()
  {
      int now = son[root][0];
      while(son[now][1])
          now = son[now][1];
      return now;
  }

SucSucSuc（得到根的后继的编号）:

  int Suc()
  {
      int now = son[root][1];
      while(son[now][0])
          now = son[now][0];
      return now;
  }

\quad\quad 模板题：洛谷 P3369 【模板】普通平衡树

\quad\quad 代码：

  #include <bits/stdc++.h>

  using namespace std;

  const int N = 100010;

  //父亲节点，两个儿子，节点个数，节点代表什么
  int fa[N], son[N][2], cnt[N], num[N] , size[N];

  int n, tot, root;

  void clear(int x)
  {
      fa[x] = son[x][0] = son[x][1] = cnt[x] = num[x] = size[x] = 0;
  }

  int find(int x)
  {
      return son[fa[x]][1] == x ? 1 : 0;
  }

  void connect(int x, int y, int w_son)
  {
      if(y) son[y][w_son] = x;
      if(x) fa[x] = y;
  }

  void push_up(int x)
  {
      size[x] = size[son[x][0]] + size[son[x][1]] + cnt[x];
  }

  void rotate(int x)
  {
      int f = fa[x], ff = fa[f] , k = find(x) , i = find(f);

      connect(son[x][k ^ 1], f, k);
      connect(f, x, k ^ 1);
      connect(x, ff, i);
      push_up(f), push_up(x);
  }

  void Splay(int x)
  {
      for(int f; (f = fa[x]) != 0; rotate(x))
          if(fa[f]) rotate(find(f) == find(x) ? f : x);

      root = x;
  }

  void Insert(int x)
  {
      if(root == 0)
          root = ++tot, size[tot] = cnt[tot] = 1, num[tot] = x;
      else
      {
          int now = root, f = 0;

          while(true)
          {
              if(num[now] == x)
              {
                  cnt[now]++;
                  push_up(now), push_up(f);
                  Splay(now);

                  return ;
              }

              f = now, now = son[now][x > num[now]];

              if(!now)
              {
                  num[++tot] = x, size[tot] = cnt[tot] = 1;
                  fa[tot] = f, son[f][x > num[f]] = tot;
                  push_up(f), Splay(tot);

                  return ;
              }
          }
      }
  }

  int Find(int x)
  {
      int now = root, ans = 0;

      while(true)
      {
          if(x < num[now])
          {
              now = son[now][0];
              continue;
          }

          ans += size[son[now][0]];

          if(num[now] == x)
          {
              Splay(now);
              return ans + 1;
          }

          ans += cnt[now];

          now = son[now][1];
      }
  }

  int Rank(int rank)
  {
      int now = root;

      while(1)
      {
          if(rank <= size[son[now][0]])
          {
              now = son[now][0];
              continue;
          }
          rank -= size[son[now][0]];

          if(rank <= cnt[now])
          {
              Splay(now);
              return num[now];
          }
          rank -= cnt[now];

          now = son[now][1];
      }
  }

  int Pre()
  {
      int now = son[root][0];
      while(son[now][1])
          now = son[now][1];
      return now;
  }

  int Suc()
  {
      int now = son[root][1];
      while(son[now][0])
          now = son[now][0];
      return now;
  }

  void Delete(int x)
  {
      Find(x);

      int now = root;

      if(cnt[now] > 1) cnt[now]--, size[now]--;
      else if(!son[now][0] && !son[now][1])
          root = 0, clear(now);
      else if(!son[now][0] && son[now][1])
          fa[root = son[now][1]] = 0, clear(now);
      else if(son[now][0] && !son[now][1])
          fa[root = son[now][0]] = 0, clear(now);
      else
      {
          int pre = Pre(), rt = root, rson = son[root][1];

          Splay(pre), connect(rson, pre, 1);
          clear(rt), push_up(root);
      }
  }

  int main()
  {
      scanf("%d", &n);
      for(int k = 1; k <= n; k++)
      {
          int opt, x;
          scanf("%d %d", &opt, &x);

          if(opt == 1)
              Insert(x);
          else if(opt == 2)
              Delete(x);
          else if(opt == 3)
              printf("%d\n", Find(x));
          else if(opt == 4)
              printf("%d\n", Rank(x));
          else if(opt == 5)
              Insert(x), printf("%d\n", num[Pre()]), Delete(x);
          else
              Insert(x), printf("%d\n", num[Suc()]), Delete(x);
      }

      return 0;
  }

\quad\quad SplaySplaySplay除了实现平衡树共有的如上功能之外，还可以处理一些区间问题（主要是区间翻转），这也是它成为LCT辅助树的最大原因...

\quad\quad 在讲解SplaySplaySplay区间处理之前，我们要明白它的一个重要性质：

\quad\quad 在以任意方式，对于任意的点，进行任意次Rotate/SplayRotate/SplayRotate/Splay操作后，SplaySplaySplay的中序遍历恒不变

\quad\quad 这是把SplaySplaySplay与区间维护联系起来的关键：SplaySplaySplay的中序遍历可以用来维护有序序列...

\quad\quad 上例题：洛谷 P3391 【模板】文艺平衡树

\quad\quad 题意很显然...让你维护一个可以区间内翻转的序列...

\quad\quad 根据观（标）察（题），不难看出，我们可以用SplaySplaySplay水过这道题...

\quad\quad 如何用SplaySplaySplay维护序列？

\quad\quad 按照之前的P3369的套路，设置一个数组d[i]d[i]d[i]表示某个点对应的权值...建立节点标号−>->−>节点权值的映射？

\quad\quad 然鹅我们发现，仅仅这么做的话，我们很难处理询问中的l,rl,rl,r...因为序列中各个点的位置信息被我们丢失了...

\quad\quad 如何解决这个问题？

\quad\quad 再做一个数组num[i]num[i]num[i]表示编号为iii的点对应的，在序列中的位置？

\quad\quad 抱歉，你会发现：在翻转的过程中维护这个数组是相当困难的...在你翻转区间后，num[i]num[i]num[i]根本就不满足BST(二叉查找树)的性质...

\quad\quad 这个地方千万不能绕进权值树的思维定式...

\quad\quad 我们注意到：在建立一棵原始SplaySplaySplay后，无论怎么旋转，每个节点在序列中的位置都等价于它在SplaySplaySplay中的排名...比如说，如果某个数在SplaySplaySplay中第k个被中序遍历到，那么它一定在原序列的第k位...

\quad\quad 这个性质在翻转后也生效，由此衍生出了另外一个性质：

\quad\quad 翻转一个序列中的[l,r][l,r][l,r]区间，相当于在SplaySplaySplay中，把排名为r+1r+1r+1 点A SplaySplaySplay到根，把排名为l−1l-1l−1的 点B SplaySplaySplay为A的左儿子，然后把B的右儿子上的所有点的左右儿子互换...

\quad\quad 一定要理解：此处SplaySplaySplay不再是一个二叉权值查找树，它不维护num[i]num[i]num[i]或者a[i]a[i]a[i]的值，但是它的中序遍历对应了原序列，它根据sizesizesize查找原序列的下标，与之前的权值SplaySplaySplay相比，每个点并不代表某个特定的维护值...

\quad\quad 最后...有几个优化...

\quad\quad 首先，我们翻转SplaySplaySplay中的区间时，可以先给要翻转的区间打上懒惰标记，然后在以后查找懒惰标记以下的点的时候，再下传懒惰标记（类似线段树）...

\quad\quad 然后，我们发现：如果按照从左往右插入原序列中的节点，那么节点的d[i]d[i]d[i]刚好等于iii，没有必要多出这么一个数组...

\quad\quad 再者，为了防止翻转时调用l−1,r+1l-1,r+1l−1,r+1越界...我们可以在一开始的时候，插入极小值与极大值（本优化会对上一优化产生影响）...

\quad\quad 最后，上代码：

#include <bits/stdc++.h>
#define ls(x) son[x][0]
#define rs(x) son[x][1]
//这么短我都define...我也真是无聊... 

using namespace std;

const int N = 100010;

int n, m , tot , root;
int son[N][2], fa[N], size[N] , rev[N];
//常规的数组...rev为懒惰标记，等于1时需要翻转 


//-----辅助函数-----//

//维护size标记 
void push_up(int x)
{
	size[x] = size[ls(x)] + size[rs(x)] + 1;
}
//下放懒惰标记 
void push_down(int x)
{
	if(rev[x]) rev[x] = 0, swap(ls(x), rs(x)), rev[ls(x)] ^= 1, rev[rs(x)] ^= 1;
}
//连接Splay中两个点 
void connect(int x, int y, int z)
{
	if(x) fa[x] = y;
	if(y) son[y][z] = x;
}
//判断某个点是父亲的左儿子还是右儿子 
int find(int x)
{
	return x == rs(fa[x]);
}



//-----主要函数-----//
 
void rotate(int x)
{
	int y = fa[x], z = fa[y], k = find(x), i = find(y);

	connect(son[x][k ^ 1], y, k);
	connect(y, x, k ^ 1);
	connect(x, z, i);

	push_up(y), push_up(x);
}

void Splay(int x, int to)
{
	for(int f; (f = fa[x]) != to; rotate(x))
		if(fa[f] != to) rotate(find(x) == find(f) ? f : x);

	if(to == 0) root = x;
}

//用插入的方法建树，当然，直接像线段树那样creat也可以 
void Insert(int x)
{
	if(!root)
		root = ++tot, size[tot] = 1;
	else
	{
		int now = root, f = 0;

		while(1)
		{
			f = now, now = rs(now) ;//后面插入的数的下标一定比前面的大，所以一路往右儿子跑就行了 

			if(!now)
			{
				size[++tot] = 1;
				rs(f) = tot, fa[tot] = f;
				push_up(f), Splay(tot, 0);

				break;
			}
		}
	}
}

//Splay中排名为k的数的编号是多少 （原序列第k位的数的编号是多少） 
int Rank(int x)
{
	int now = root;

	while(1)
	{
		//不要忘了下放标记 
		push_down(now);

		if(x <= size[ls(now)])
		{
			now = ls(now);

			continue;
		}
		x -= size[ls(now)];

		if(x == 1)
			return now;
		x--;

		now = rs(now);
	}
}

//翻转函数 
void Reverse(int l, int r)
{
	//先找到两个边界...
	//因为在之前插入了+INF和-INF的原因，这里处理有些特殊 
	int L = Rank(l), R = Rank(r + 2);

	//两次Splay 
	Splay(L, 0), Splay(R, L);
	//打标记 
	rev[ls(R)] ^= 1;
}

//最后输出时的中序遍历 
void Dfs(int rt)
{
	if(!rt) return ;

	//不要忘了下放标记 
	push_down(rt);

	Dfs(ls(rt));
	//这里的处理还是因为+INF和-INF... 
	if(rt >= 2 && rt <= n + 1)
		printf("%d ", rt - 1);
	Dfs(rs(rt));
}

int main()
{
	scanf("%d %d", &n, &m);

	//0xefefefef代表-INF 
	Insert(0xefefefef);
	for(int k = 1; k <= n; k++)
		Insert(k);
	//0x3f3f3f3f代表+INF 
	Insert(0x3f3f3f3f);

	for(int k = 1; k <= m; k++)
	{
		int l, r;
		scanf("%d %d", &l, &r);

		Reverse(l, r);
	}

	Dfs(root);

	return 0;
}
